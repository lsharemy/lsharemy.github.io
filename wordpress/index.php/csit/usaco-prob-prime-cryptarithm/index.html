int isgooddigit[10] = {0};

bool isgoodnum(int num, int d)
{
int cnt = 0;
while (num)
{
if(!isgooddigit[num%10]) return false;
num /= 10;
cnt++;
}
if (cnt != d) return false;
return true;
}

bool issolution(int i, int j)
{
if (!isgoodnum(i, 3) || !isgoodnum(j, 2) || !isgoodnum(ij, 4)) return false;
while (j)
{
if (!isgoodnum(i(j%10), 3)) return false;
j /= 10;
}
return true;
}

ofstream fout ("crypt1.out");
ifstream fin ("crypt1.in");
int N;
fin >> N;
for (int i = 0; i < N; i++)
{
int digit;

isgooddigit[digit] = 1;
}
int res = 0;
for (int i = 100; i < 1000; i++)
{
for (int j = 10; j < 100; j++)
{

}
}
fout << res << endl;
