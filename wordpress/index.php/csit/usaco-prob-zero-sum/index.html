/ evaluate the string s as arithmetic and return the sum /
int
eval(char s)
{
int term, sign, sum;
char p;

sign = 1;
term = 0;
sum = 0;
for(p=s; *p; p++) {
    switch(*p){
    case '+':
    case '-':
        sum += sign*term;
        term = 0;
        sign = *p == '+' ? 1 : -1;
        break;
    case ' ':
        break;
    default:    /* digit */
        term = term*10 + *p-'0';
    }
}
sum += sign*term;
return sum;

}
